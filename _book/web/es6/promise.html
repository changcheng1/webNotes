  import {
  setTimeout
} from "timers";

{
  // 定义函数
  var ajax = function () {
    console.log('1')
    return new Promise(function (resolve, reject) {
      setTimeout(() => {
        resolve("")
      }, 1000);
    })
  }
  // 执行函数
  ajax().then(function () {
    return new Promise((resolve, reject) => {
      console.log("2")
      resolve("")
    })
  }).then(function () {
    // 返回一个promise实例
    return new Promise((resolve, reject) => {
      console.log("3")
      // resolve:表示允许下一步
      resolve("")
    })
  }).then(function () {
    console.log("4")
  })
} {
  let ajax = function (num) {
    return new Promise(function (resolve, reject) {
      if (num > 6) {
        resolve()
      } else {
        throw new Error("Num小于6")
      }
    })
  }
  ajax(7).then(function () {
    console.log("then", "大于6")
  }).catch(function (err) {
    console.log("catch", err)
  })
  ajax(5).then(function () {
    console.log("then", "大于6")
  }).catch(function (err) {
    // catch中err用于获取传参中的err
    console.log("catch", err)
  })

  let imgLoad = function (src) {
    return new Promise((resolve, reject) => {
      let img = new Image();
      img.src = src;
      img.width = 300;
      img.height = 300;
      img.onload = function () {
        setTimeout(() => {
          // 传img进去
          resolve(img);
        }, 1000)
      }
      img.onerror = function (err) {
        reject(err);
      }
    })
  }
  let addImg = function (images) {
    images.forEach(img => {
      document.body.appendChild(img)
    })
  }
  Promise.all([
